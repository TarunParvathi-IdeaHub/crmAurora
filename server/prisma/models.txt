
model ProgramFeeStructure {
  id String @id @default(uuid())

  institutionId String
  institution Institution @relation(fields: [institutionId], references: [id])

  programId String
  program Program @relation(fields: [programId], references: [id])

  academicYear String

  tuitionFee Float
  admissionFee Float
  specialFee Float?
  placementTrainingFee Float?
  studentExpanses Float?

  isActive Boolean @default(true)

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  @@unique([institutionId, programId, academicYear])

  @@map("programFeeStructures")
}

enum FeeCategoryType {
  APPLICATION_FEE
  REGISTRATION_FEE
  TUITION_FEE
  HOSTEL_FEE
  TRANSPORT_FEE
  EXAM_FEE
}

model FeeCategory {
  id String @id @default(uuid())

  institutionId String
  institution Institution @relation(fields: [institutionId], references: [id])

  categoryType FeeCategoryType

  feeName String
  amount Float

  isFixed Boolean @default(true)

  isActive Boolean @default(true)

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  @@unique([institutionId, categoryType])
  invoices Invoice[]

  @@map("feeCategories")
}

enum InvoiceStatus {
  PENDING
  PARTIALLY_PAID
  PAID
  OVERDUE
  CANCELLED
}

model Invoice {
  id String @id @default(uuid())

  institutionId String
  institution Institution @relation(fields: [institutionId], references: [id])

  invoiceNumber String @unique

  invoiceDate DateTime
  dueDate DateTime?

  studentAdmissionApplicationId String?
  studentAdmissionApplication StudentAdmissionApplication? @relation(fields: [studentAdmissionApplicationId], references: [id])

  feeCategoryId String
  feeCategory FeeCategory @relation(fields: [feeCategoryId], references: [id])

  amount Float

  status InvoiceStatus @default(PENDING)

  remarks String?

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  paymentTransactions PaymentTransaction[]
  applicationFees ApplicationFee[]
  registrationFees RegistrationFee[]

  @@map("invoices")
}
